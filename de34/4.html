import random
import json
from datetime import datetime
from flask import Flask, render_template, request
from openai import OpenAI

app = Flask(__name__)

client = OpenAI(
    api_key=""
)

ZODIACS = [
    "牡羊座","牡牛座","双子座","蟹座",
    "獅子座","乙女座","天秤座","蠍座",
    "射手座","山羊座","水瓶座","魚座"
]


def get_seed():
    return datetime.now().strftime("%Y-%m-%d")


def generate_fortune(zodiac, seed):

    prompt = f"""
あなたは魔法の占い師です。

{zodiac}の今日の運勢を以下のJSON形式で返してください：

{{
  "love": "恋愛運（100文字以内）",
  "money": "金運（100文字以内）",
  "study": "勉強運（100文字以内）",
  "lucky_color": "ラッキーカラー",
  "lucky_item": "ラッキーアイテム"
}}

必ず日本語で、ポジティブにしてください。
"""

    random.seed(seed + zodiac)

    res = client.chat.completions.create(
        model="gpt-4.1-mini",
        messages=[{"role": "user", "content": prompt}]
    )

    text = res.choices[0].message.content

    # JSON化（多少崩れても対応）
    try:
        return json.loads(text)
    except:
        return {
            "love": text,
            "money": "",
            "study": "",
            "lucky_color": "不明",
            "lucky_item": "不明"
        }


def generate_ranking(seed):
    random.seed(seed)
    shuffled = ZODIACS.copy()
    random.shuffle(shuffled)

    return [{"rank": i+1, "zodiac": z} for i, z in enumerate(shuffled)]


@app.route("/", methods=["GET", "POST"])
def index():
    seed = get_seed()
    selected = None
    fortune = None
    ranking = None

    if request.method == "POST":
        selected = request.form.get("zodiac")
        fortune = generate_fortune(selected, seed)
        ranking = generate_ranking(seed)

    return render_template(
        "index.html",
        zodiacs=ZODIACS,
        selected=selected,
        fortune=fortune,
        ranking=ranking,
        today=seed
    )


if __name__ == "__main__":
    app.run(
        host="0.0.0.0",
        port=3000,
        debug=True
    )
